\poemtitle{The cow}

\begin{verse}
    There was a cow. \\
    The cow got up and walked away. \\
    It was afraid.
\end{verse}
\begin{verse}
    The crocodile continuted to stalk her with lustful eyes. \\
    She stopped chewing when she realised her hooves are stepping into mud. \\
    This is a swamp.
\end{verse}
\begin{verse}
    The cow panicked and tried to jump. \\
    Cows cannot jump.
\end{verse}
\begin{verse}
    A crocodile ate a cow.
\end{verse}
\attrib{6th of February, 2018}

\poemtitle{Tangled}

\begin{verse}
Our feelings are like \\
\breathe a ball of twine.
\end{verse}

\begin{verse}
Tied to our head.
\end{verse}

\begin{verse}
We won't get rid of them, \\
and we shouldn't try.
\end{verse}

\begin{verse}
We can, though, for a start, \\
\breathe untangle them.
\end{verse}
\attrib{Wellbeing workshop, 12th of March, 2018}

\poemtitle{Piggy}

\begin{verse}
    There's an innocence \\
    \breathe that can only be had, \\
    By a piglet rolling through mud.
\end{verse}

\begin{verse}
    The burdens we carry, \\
    \breathe tied to our heads, \\
    Will only be lost, \\
    If we tickle ourselves.
\end{verse}
\attrib{Wellbeing workshop, 12th of March, 2018}
